    <!-- 🧠 ALL JAVASCRIPT DATABASE & LOGIC LIVES HERE NOW -->
    <script>
        const chemicalDatabase = {
            "Medicine": {
                "bicaridine": { name: "Bicaridine", ingredients: [{amount: 1, name: "carbon"}, {amount: 1, name: "oxygen"}], recipeText: "Mix to create baseline brute healing medicine." }
            },
            "Special": {
                "plastic polymers": { name: "Plastic polymers", ingredients: [{amount: 1, name: "oil"}, {amount: 1, name: "plasticizer"}] }
            },
            "Other": {
                "salt": { name: "Salt", ingredients: [{amount: 1, name: "sodium"}, {amount: 1, name: "chlorine"}] }
            }
        };

        let currentCategory = 'Medicine';
        let searchQuery = '';

        function initRecipeWebsite() {
            renderTabs();
            renderRecipes();
            setupSearch();
        }

        function renderTabs() {
            const tabContainer = document.getElementById('category-tabs');
            if (!tabContainer) return;
            tabContainer.innerHTML = '';
            
            Object.keys(chemicalDatabase).forEach(category => {
                const button = document.createElement('button');
                button.textContent = category;
                button.className = `tab-btn ${category === currentCategory ? 'active' : ''}`;
                button.addEventListener('click', () => {
                    currentCategory = category;
                    document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
                    button.classList.add('active');
                    renderRecipes();
                });
                tabContainer.appendChild(button);
            });
        }

        function setupSearch() {
            const searchInput = document.getElementById('recipe-search');
            if (searchInput) {
                searchInput.addEventListener('input', (e) => {
                    searchQuery = e.target.value.toLowerCase().trim();
                    renderRecipes();
                });
            }
        }

        function renderRecipes() {
            const displayGrid = document.getElementById('recipe-grid');
            if (!displayGrid) return;
            displayGrid.innerHTML = '';

            const activeCategoryData = chemicalDatabase[currentCategory] || {};
            
            Object.keys(activeCategoryData).forEach(key => {
                const item = activeCategoryData[key];
                const matchesName = item.name.toLowerCase().includes(searchQuery);
                const matchesIngredients = item.ingredients && item.ingredients.some(ing => ing.name.toLowerCase().includes(searchQuery));

                if (searchQuery && !matchesName && !matchesIngredients) return;

                const card = document.createElement('div');
                card.className = 'recipe-card';

                let cardHTML = `<h3>${item.name}</h3>`;
                if (item.ingredients) {
                    cardHTML += `<ul class="ingredient-list">`;
                    item.ingredients.forEach(ing => {
                        cardHTML += `<li><strong>${ing.amount}u</strong> ${ing.name}</li>`;
                    });
                    cardHTML += `</ul>`;
                }
                if (item.recipeText) {
                    cardHTML += `<div class="recipe-notes">${item.recipeText}</div>`;
                }

                card.innerHTML = cardHTML;
                displayGrid.appendChild(card);
            });
        }

        document.addEventListener('DOMContentLoaded', initRecipeWebsite);
    </script>
